126# safely modify our conf file; write its contents to a temp file, modify the
127# temp file, write the contents of the temp file back to the conf file
128# TODO: if multiple options are provided to this script, this sub should write
129# them all at once, instead of only a single one and having to call it once for
130# each option specified to the script.
131sub config_write {
132        my ($key, $val) = @_;
133        @_ == 2 or script_error('config_write requires two arguments.');
134        if (! -d $conf_dir) {
135                mkdir $conf_dir or usage_error("Unable to create $conf_dir. Exiting.");
136        }
137        if (-f $conf_file) {
138                my $tempfh = tempfile(DIR => $tempdir);
139                my ($conffh, $exit) = open_read($conf_file);
140                if ($exit) {
141                        warn $conffh;
142                        exit $exit;
143                }
144                my $conftents = do {local $/; <$conffh>};
145                print {$tempfh} $conftents;
146                # tie the temp file so that if $key is already there, we just change
147                # that line and untie it
148                tie my @temp, 'Tie::File', $tempfh;
149                my $has;
150                my $regex = qr/\A\Q$key\E=/;
151                # TODO: fix this monstrosity
152                FIRST: for my $tmpline (@temp) {
153                        $has++, $tmpline = "$key=$val", last FIRST if $tmpline =~ $regex;
154                }
155                untie @temp;
156                # otherwise, append our new $key=$value pair
157                print {$tempfh} "$key=$val\n" unless $has;
158                # then overwrite the conf file with the contents of the temp file
159                seek $tempfh, 0, 0;
160                my $contents = do {local $/; <$tempfh>};
161                close $conffh;
162                ($conffh, $exit) = open_fh($conf_file, '>');
163                if ($exit) {
164                        warn $conffh;
165                        exit $exit;
166                }
167                print {$conffh} $contents or return();
168        } else {
169                # no config file, easiest case of all.
170                my ($fh, $exit) = open_fh($conf_file, '>') or return();
171                if ($exit) {
172                        warn $fh;
173                        exit $exit;
174                }
175                print {$fh} "$key=$val\n";
176                close $fh;
177        }
178        return 1;
179}
