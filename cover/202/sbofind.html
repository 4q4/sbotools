65sub perform_search {

109                                                next FIRST;
110                                        } else {
111                                                redo LOCATION; # But if it isn't, we try again...
112                                        }
113                                }
114                        }
115                }
116        }
117        if ($config{LOCAL_OVERRIDES} ne 'FALSE') {

119                while (my $dir = readdir($dh)) {
120                        next if $local{$dir};

123                        }
124                }

126        }
127        return \@findings;
